#include<iostream>
#include<string>
using namespace std;
int readgrads(int from,int to)
{
	int grade;
	do
	{
		cout << "please entre a grade between 0 and 100:" << endl;
		cin >> grade;
	} while (grade<from || grade>to);
	return grade;
}
char getgradelettre(int grade)
{
	if (grade >= 90)
		return 'A';
	else if (grade >= 80)
		return 'B';
	else if (grade >= 70)
		return 'C';
	else if (grade >= 60)
		return 'D';
	else if (grade >= 50)
		return 'E';
	else 
		return 'F';
}
int main()
{
	cout << readgrads(0, 100) << endl;
	cout << "your grade is: " << getgradelettre(0.100) << endl;
		return 0;
}
